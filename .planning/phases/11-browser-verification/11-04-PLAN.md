Fix GAP-06 (drag-and-drop drops ON stop instead of BETWEEN).

**Current behavior:** Each stop card has `ondragover` and `ondrop` handlers directly on the card. When you drag stop A onto stop B, it triggers `_onStopDrop(event, B_index)` which swaps A and B. But visually it looks like you're dropping ON top of B, not inserting BEFORE B.

**Fix approach:** Add thin drop-zone divs between each stop card. Remove drag/drop handlers from the cards themselves. Only the drop zones handle `ondrop`.

**Step 1 — Update `renderStopsOverview` in guide-stops.js (~line 78-84):**

Replace the current implementation with one that interleaves drop zones between cards. The function currently returns HTML that wraps cards in a `stop-cards-list` div followed by additional HTML (the add-stop button). Preserve whatever follows the cards list in the existing return statement — read the full function to see the complete return value before editing.

```javascript
function renderStopsOverview(plan) {
  var stops = plan.stops || [];
  var html = '';
  for (var i = 0; i < stops.length; i++) {
    // Drop zone before each card (insert at position i)
    html += '<div class="stop-drop-zone" data-drop-index="' + i + '" ' +
      'ondragover="event.preventDefault(); this.classList.add(\'drop-zone-active\')" ' +
      'ondragleave="this.classList.remove(\'drop-zone-active\')" ' +
      'ondrop="_onDropZoneDrop(event, ' + i + ')">' +
      '<div class="drop-zone-line"></div></div>';
    html += renderStopCard(stops[i], i, stops.length);
  }
  // Drop zone after last card (insert at end = stops.length)
  html += '<div class="stop-drop-zone" data-drop-index="' + stops.length + '" ' +
    'ondragover="event.preventDefault(); this.classList.add(\'drop-zone-active\')" ' +
    'ondragleave="this.classList.remove(\'drop-zone-active\')" ' +
    'ondrop="_onDropZoneDrop(event, ' + stops.length + ')">' +
    '<div class="drop-zone-line"></div></div>';

  // Wrap cards+zones in list div, then append the rest of the existing return
  // (e.g., the add-stop button HTML). Read the full original return statement
  // and preserve everything after the cards list wrapper.
  return '<div class="stop-cards-list">' + html + '</div>' + /* ...existing add-stop button HTML unchanged... */;
}
```

**Step 3 — Add `_onDropZoneDrop` handler in guide-edit.js (after `_onStopDrop`):**

This function MUST be declared at top-level module scope (not inside an IIFE or closure) so that inline HTML `ondrop` attributes can call it.

```javascript
async function _onDropZoneDrop(e, dropBeforeIndex) {
  e.preventDefault();
  document.querySelectorAll('.stop-drop-zone').forEach(z => z.classList.remove('drop-zone-active'));

  if (_dragStopSourceIndex === null) return;
  var sourceIdx = _dragStopSourceIndex;

  // Convert "insert before dropBeforeIndex" to "move to position"
  // If dragging downward, the removal of source shifts indexes
  var targetIdx = dropBeforeIndex;
  if (sourceIdx < dropBeforeIndex) targetIdx = dropBeforeIndex - 1;

  if (sourceIdx === targetIdx) {
    _dragStopSourceIndex = null;
    return;
  }

  // Delegate to existing _onStopDrop
  _onStopDrop(e, targetIdx);
}
```

```javascript
function _editStopNights(stopId, currentNights) {
  if (_editInProgress) return;
  var input = prompt('Anzahl Naechte:', currentNights);
  if (input === null) return;
  var nights = parseInt(input);
  if (isNaN(nights) || nights < 1 || nights > 14) {
    alert('Bitte eine Zahl zwischen 1 und 14 eingeben.');
    return;
  }
  if (nights === currentNights) return;

  // Update local state only — persisted when travel is saved
  var stops = (S.result && S.result.stops) || [];
  for (var i = 0; i < stops.length; i++) {
    if (stops[i].id === stopId) {
      stops[i].nights = nights;
      break;
    }
  }

  // Update localStorage cache
  var travelId = S.result._saved_travel_id || S.result.id;
  lsSet(LS_RESULT, { savedAt: new Date().toISOString(), plan: S.result });

  // Re-render stops view to show updated nights
  renderGuide(S.result, 'stops');
}
```

**NOTE:** This is a local-state-only edit. The nights change lives in `S.result` and localStorage until the full travel plan is saved via the existing save mechanism. A dedicated backend PATCH endpoint for individual stop nights is out of scope for this gap closure fix.
