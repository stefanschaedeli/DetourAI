```javascript
function addPreferredTagFromInput() {
  const inp = document.getElementById('preferred-tag-input');
  const val = inp.value.trim();
  if (!val || S.preferredTags.includes(val)) { inp.value = ''; return; }
  S.preferredTags.push(val);
  inp.value = '';
  renderPreferredTags();
  saveFormToCache();
}

function renderPreferredTags() {
  const c = document.getElementById('preferred-tags');
  if (!c) return;
  // Note: esc() is used for XSS prevention on user content per project convention
  c.textContent = '';
  S.preferredTags.forEach((t, i) => {
    const span = document.createElement('span');
    span.className = 'tag-chip';
    span.textContent = t + ' ';
    const btn = document.createElement('button');
    btn.textContent = '\u00d7';
    btn.onclick = () => removePreferredTag(i);
    span.appendChild(btn);
    c.appendChild(span);
  });
}

function removePreferredTag(idx) {
  S.preferredTags.splice(idx, 1);
  renderPreferredTags();
  saveFormToCache();
}
```
