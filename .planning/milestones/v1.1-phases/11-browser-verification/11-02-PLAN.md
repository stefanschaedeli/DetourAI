Fix GAP-02 (map over-zooms on day drill-down) in frontend/js/maps.js.

The `fitDayStops` function (line ~805) calls `_guideMap.fitBounds(bounds, padding)` without a maxZoom constraint. When a day has stops close together, Google Maps zooms in too far and markers/POIs become invisible.

Fix: After the `fitBounds` call, add a maxZoom listener that caps the zoom level at 13. This is the standard Google Maps pattern:

```javascript
function fitDayStops(stops) {
    if (!_guideMap || !stops.length) return;
    var bounds = new google.maps.LatLngBounds();
    stops.forEach(function(s) {
      if (s.lat && s.lng) bounds.extend({ lat: s.lat, lng: s.lng });
    });
    _guideMap.fitBounds(bounds, { top: 48, right: 48, bottom: 48, left: 48 });
    // Cap zoom to prevent over-zooming on nearby stops (GAP-02)
    var zoomListener = google.maps.event.addListenerOnce(_guideMap, 'idle', function() {
      if (_guideMap.getZoom() > 13) _guideMap.setZoom(13);
    });
}
```

The `addListenerOnce` fires after fitBounds animation completes and self-removes. Zoom level 13 shows city/neighborhood detail with visible markers and POIs.
