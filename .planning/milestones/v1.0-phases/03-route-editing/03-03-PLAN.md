POST /api/travels/{travel_id}/remove-stop
  Request: { stop_id: int }
  Response: { job_id: string, status: "editing" }

POST /api/travels/{travel_id}/add-stop
  Request: { location: string, insert_after_stop_id: int, nights: int }
  Response: { job_id: string, status: "editing" }
  Errors: 400 (empty location, geocode fail, stop not found), 404, 409

POST /api/travels/{travel_id}/reorder-stops
  Request: { old_index: int, new_index: int }
  Response: { job_id: string, status: "editing" }
  Errors: 400 (invalid/same indices), 404, 409

POST /api/travels/{travel_id}/replace-stop (enhanced)
  Request: { stop_id: int, mode: string, manual_location?: string, manual_nights?: int, hints?: string }

SSE Events:

  add_stop_progress: { phase: string, message: string }
  add_stop_complete: full plan object
